pipeline{
    agent {
        label 'node1'
    }
    tools{
        maven 'maven3'
    }
    parameters {
        choice choices: ['prod', 'dev'], name: 'server_env'
    }
    environment{
        NAME='amal'
    }
    
    stages{
        stage('build'){
            steps{
                sh 'mvn clean package -DskipTests'
            }
            post{
                success{
                    archiveArtifacts artifacts: "**/target/*.war"
                }
            }
        }
        stage('deployee'){
            steps{
                    echo "this is to ${params.server_env}"
                }
        }
        stage("stashing"){
            steps{
                sh "mvn test"
                dir("target/"){
                   stash name:"build-war",includes:"*.war" 
                }

            }
        }
        stage("unstash"){
            when{expression{params.select_env=="prod"}}
            steps{
                sh "mkdir unstash"
                script{
                    unstash "build-war"
                    sh "mv *.war unstash/"
                }
            }
        }
        stage('test'){
            parallel{
                stage('testA'){
                    agent {
                        label 'node1'
                    }
                    steps{
                        echo "this is stage A"
                    }
                }
                stage('testB'){
                    steps{
                        echo "this is stage B"
                    }
                }

 
            }
        }

    }
    
}


stash and unstash
