pipeline{
    agent {
        label 'node1'
    }
    tools{
        maven 'maven3'
    }
    parameters {
        choice choices: ['prod', 'dev'], name: 'server_env'
    }
    environment{
        NAME='amal'
    }
    
    stages{
        stage('build'){
            steps{
                sh 'mvn clean package'
            }
            post{
                success{
                    archiveArtifacts artifacts: "**/target/*.war"
                }
            }
        }
        stage('deployee'){
            steps{
                    echo "this is to ${params.server_env}"
                }
        }
        stage('test'){
            parallel{
                stage('testA'){
                    agent {
                        label 'node1'
                    }
                    steps{
                        echo "this is stage A"
                    }
                }
                stage('testB'){
                    steps{
                        echo "this is stage B"
                    }
                }

 
            }
        }

    }
    
}
